def keys_values_and_items():
     S={'a':30, 'b':40, 'c':50, 'd':60}
     S.keys()
     for y in S.keys():
           print(y)
     P={'a':120, 'b':340, 'c':650, 'd':860}
     P.values()
     for z in P.values():
           print(z)
     A={'a':30, 'b':40, 'c':50, 'd':60}
     A.items()
     for t in A.items():
           print(t)
